454  static shape_t
455  validate_broadcast(const shape_t& lhs, const shape_t& rhs)
456  {
457  // This ensures that LHS is always the bigger vector.
458  if (lhs.size() < rhs.size())
459  return validate_broadcast(rhs, lhs);
460 
461  shape_t result_shape(lhs.size());
462  size_t l = lhs.size();
463 
464  for (size_t r = rhs.size(); r >= 0; --r, --l)
465  {
466  if (rhs[r] != lhs[l] && rhs[r] > 1 && lhs[l] > 1)
467  {
468  std::cout << "ERROR - broadcast - 1" << std::endl;
469  // throw error.
470  }
471 
472  result_shape[l] = std::max(rhs[r], lhs[l]);
473  }
474 
475  while (l >= 0)
476  result_shape[l--] = lhs[l];
477 
478  return result_shape;
479  }

511  template<typename lhs_t, typename rhs_t, typename res_t>
512  static NdArray<res_t>
513  broadcast(
514  const NdArray<lhs_t>& lhs,
515  const NdArray<rhs_t>& rhs,
516  std::function<res_t(lhs_t, rhs_t)>& element_op)
517  {
518  auto result_shape =
519  validate_broadcast(lhs.shape(), rhs.shape());
520  auto result = NdArray<res_t>(result_shape);
521 
522  auto lhs_shape = lhs.shape();
523  auto rhs_shape = rhs.shape();
524  auto n_lhs = lhs_shape.size();
525  auto n_rhs = rhs_shape.size();
526  auto n_res = result_shape.size();
527 
534  std::function<void(size_t, size_t, size_t, size_t)>
535  broadcast_helper =
536  [&](size_t axis,size_t lhs_i, size_t rhs_i, size_t res_i)
537  {
538  size_t lhs_pre = lhs_i;
539  size_t rhs_pre = rhs_i;
540  size_t res_pre = res_i;
541 
542  for (size_t i = 0; i < result_shape[axis]; ++i)
543  {
544  // Ensures that index is not updated in case the
545  // right- based shape vector hasn't begun yet.
546  // Indices would begin once n_res - axis is
547  // equal to n_l.
548  // Correspondin index 0 would be given by
549  // n_l - (n_res - axis) or n_l + axis - n_res.
550  // The reason this is checked for both is
551  // because either could be the smaller array.
552  if (n_lhs + axis >= n_res)
553  {
554  // The above condition would always be true
555  // for the larger array.
556  // Thus, the following ternary to ensure
557  // that the index formular is used only if
558  // array is smaller.
559  auto axis_l =
560  n_res > n_lhs ?
561  n_lhs + axis - n_res : axis;
562  lhs_i =
563  lhs_pre +
564  (i % lhs_shape[axis_l]) *
565  lhs.strides()[axis_l];
566  }
567 
568  if (n_rhs + axis >= n_res)
569  {
570  auto axis_r =
571  n_res > n_rhs ?
572  n_rhs + axis - n_res : axis;
573  rhs_i =
574  rhs_pre +
575  (i % rhs_shape[axis_r]) *
576  rhs.strides()[axis_r];
577  }
578 
579  res_i = res_pre + i * result.strides()[axis];
580 
581  // Recurse until the final axis is reached.
582  // Perform the element operation then.
583  if (n_res - 1 > axis)
584  broadcast_helper(
585  axis + 1,
586  lhs_i,
587  rhs_i,
588  res_i);
589  else
590  result.set(
591  res_i,
592  element_op(
593  lhs.get(lhs_i),
594  rhs.get(rhs_i)));
595  }
596  };
597 
598  broadcast_helper(0, 0, 0, 0);
599 
600  return result;
601  }
