482  static shape_t
483  validate_broadcast(const shape_t& lhs, const shape_t& rhs)
484  {
485  // This ensures that LHS is always the bigger vector.
486  if (lhs.size() < rhs.size())
487  return validate_broadcast(rhs, lhs);
488 
489  shape_t result_shape(lhs.size());
490  size_t l = lhs.size();
491 
492  for (size_t r = rhs.size(); r >= 0; --r, --l)
493  {
494  if (rhs[r] != lhs[l] && rhs[r] > 1 && lhs[l] > 1)
495  {
496  std::cout << "ERROR - broadcast - 1" << std::endl;
497  // throw error.
498  }
499 
500  result_shape[l] = std::max(rhs[r], lhs[l]);
501  }
502 
503  while (l >= 0)
504  result_shape[l--] = lhs[l];
505 
506  return result_shape;
507  }

539  template<typename lhs_t, typename rhs_t, typename res_t>
540  static NdArray<res_t>
541  broadcast(
542  const NdArray<lhs_t>& lhs,
543  const NdArray<rhs_t>& rhs,
544  std::function<res_t(lhs_t, rhs_t)>& element_op)
545  {
546  auto result_shape =
547  validate_broadcast(lhs.shape(), rhs.shape());
548  auto result = NdArray<res_t>(result_shape);
549 
550  auto lhs_shape = lhs.shape();
551  auto rhs_shape = rhs.shape();
552  auto n_lhs = lhs_shape.size();
553  auto n_rhs = rhs_shape.size();
554  auto n_res = result_shape.size();
555 
562  std::function<void(size_t, size_t, size_t, size_t)>
563  broadcast_helper =
564  [&](size_t axis,size_t lhs_i, size_t rhs_i, size_t res_i)
565  {
566  size_t lhs_pre = lhs_i;
567  size_t rhs_pre = rhs_i;
568  size_t res_pre = res_i;
569 
570  for (size_t i = 0; i < result_shape[axis]; ++i)
571  {
572  // Ensures that index is not updated in case the
573  // right- based shape vector hasn't begun yet.
574  // Indices would begin once n_res - axis is
575  // equal to n_l.
576  // Correspondin index 0 would be given by
577  // n_l - (n_res - axis) or n_l + axis - n_res.
578  // The reason this is checked for both is
579  // because either could be the smaller array.
580  if (n_lhs + axis >= n_res)
581  {
582  // The above condition would always be true
583  // for the larger array.
584  // Thus, the following ternary to ensure
585  // that the index formular is used only if
586  // array is smaller.
587  auto axis_l =
588  n_res > n_lhs ?
589  n_lhs + axis - n_res : axis;
590  lhs_i =
591  lhs_pre +
592  (i % lhs_shape[axis_l]) *
593  lhs.strides()[axis_l];
594  }
595 
596  if (n_rhs + axis >= n_res)
597  {
598  auto axis_r =
599  n_res > n_rhs ?
600  n_rhs + axis - n_res : axis;
601  rhs_i =
602  rhs_pre +
603  (i % rhs_shape[axis_r]) *
604  rhs.strides()[axis_r];
605  }
606 
607  res_i = res_pre + i * result.strides()[axis];
608 
609  // Recurse until the final axis is reached.
610  // Perform the element operation then.
611  if (n_res - 1 > axis)
612  broadcast_helper(
613  axis + 1,
614  lhs_i,
615  rhs_i,
616  res_i);
617  else
618  result.set(
619  res_i,
620  element_op(
621  lhs.get(lhs_i),
622  rhs.get(rhs_i)));
623  }
624  };
625 
626  broadcast_helper(0, 0, 0, 0);
627 
628  return result;
629  }
